66 class AuthorizedUserCredentials : public Credentials {

73  explicit AuthorizedUserCredentials(std::string const& content,
74  std::string const& source,
75  std::string oauth_server)
76  : expiration_time_() {
77  HttpRequestBuilderType request_builder(
78  std::move(oauth_server),
79  storage::internal::GetDefaultCurlHandleFactory());
80  auto info = ParseAuthorizedUserCredentials(content, source);
81  std::string payload("grant_type=refresh_token");
82  payload += "&client_id=";
83  payload += request_builder.MakeEscapedString(info.client_id).get();
84  payload += "&client_secret=";
85  payload += request_builder.MakeEscapedString(info.client_secret).get();
86  payload += "&refresh_token=";
87  payload += request_builder.MakeEscapedString(info.refresh_token).get();
88  payload_ = std::move(payload);
89  request_ = request_builder.BuildRequest();
90  }

92  std::string AuthorizationHeader() override {
93  std::unique_lock<std::mutex> lk(mu_);
94  cv_.wait(lk, [this]() { return Refresh().ok(); });
95  return authorization_header_;
96  }
97 
98  private:
99  storage::Status Refresh() {
100  namespace nl = storage::internal::nl;
101  if (std::chrono::system_clock::now() < expiration_time_) {
102  return storage::Status();
103  }
104 
105  // TODO(#516) - use retry policies to refresh the credentials.
106  auto response = request_.MakeRequest(payload_);
107  if (response.status_code >= 300) {
108  return storage::Status(response.status_code, std::move(response.payload));
109  }
110  nl::json access_token = nl::json::parse(response.payload, nullptr, false);
111  if (access_token.is_discarded() or
112  access_token.count("access_token") == 0U or
113  access_token.count("expires_in") == 0U or
114  access_token.count("id_token") == 0U or
115  access_token.count("token_type") == 0U) {
116  return storage::Status(
117  response.status_code, std::move(response.payload),
118  "Could not find all required fields in response (access_token,"
119  " id_token, expires_in, token_type).");
120  }
121  std::string header = "Authorization: ";
122  header += access_token.value("token_type", "");
123  header += ' ';
124  header += access_token.value("access_token", "");
125  std::string new_id = access_token.value("id_token", "");
126  auto expires_in =
127  std::chrono::seconds(access_token.value("expires_in", int(0)));
128  auto new_expiration = std::chrono::system_clock::now() + expires_in -
129  GoogleOAuthAccessTokenExpirationSlack();
130  // Do not update any state until all potential exceptions are raised.
131  authorization_header_ = std::move(header);
132  expiration_time_ = new_expiration;
133  return storage::Status();
134  }
