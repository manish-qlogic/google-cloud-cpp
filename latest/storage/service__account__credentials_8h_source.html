55 template <typename HttpRequestBuilderType =
56  storage::internal::CurlRequestBuilder,
57  typename ClockType = std::chrono::system_clock>
58 class ServiceAccountCredentials : public Credentials {
59  public:
60  explicit ServiceAccountCredentials(std::string const& content)
61  : ServiceAccountCredentials(content, GoogleOAuthRefreshEndpoint()) {}
62 
63  explicit ServiceAccountCredentials(std::string const& content,
64  std::string default_token_uri)
65  : expiration_time_(), clock_() {
66  namespace nl = storage::internal::nl;
67  nl::json credentials = nl::json::parse(content);
68  // Below, we construct a JWT refresh request used to obtain an access token.
69  // The structure of a JWT is defined in RFC 7519 (see
70  // https://tools.ietf.org/html/rfc7519), and Google-specific JWT validation
71  // logic is further described at:
72  // https://cloud.google.com/endpoints/docs/frameworks/java/troubleshoot-jwt
73  nl::json assertion_header = {
74  {"alg", "RS256"},
75  {"kid", credentials["private_key_id"].get_ref<std::string const&>()},
76  {"typ", "JWT"}};
77 
78  std::string scope = GoogleOAuthScopeCloudPlatform();
79  // Some credential formats (e.g. gcloud's ADC file) don't contain a
80  // "token_uri" attribute in the JSON object. In this case, we try using the
81  // default value.
82  char const TOKEN_URI_KEY[] = "token_uri";
83  std::string token_uri = credentials.value(TOKEN_URI_KEY, default_token_uri);
84  long int cur_time = static_cast<long int>(
85  std::chrono::system_clock::to_time_t(clock_.now()));
86  long int expiration_time =
87  cur_time + GoogleOAuthAccessTokenLifetime().count();
88  nl::json assertion_payload = {
89  {"iss", credentials["client_email"].get_ref<std::string const&>()},
90  {"scope", scope},
91  {"aud", token_uri},
92  {"iat", cur_time},
93  // Resulting access token should be expire after one hour.
94  {"exp", expiration_time}};
95 
96  HttpRequestBuilderType request_builder(
97  std::move(token_uri), storage::internal::GetDefaultCurlHandleFactory());
98  std::string svc_acct_private_key_pem =
99  credentials["private_key"].get_ref<std::string const&>();
100  // This is the value of grant_type for JSON-formatted service account
101  // keyfiles downloaded from Cloud Console.
102  std::string payload("grant_type=");
103  payload +=
104  request_builder
105  .MakeEscapedString("urn:ietf:params:oauth:grant-type:jwt-bearer")
106  .get();
107  payload += "&assertion=";
108  payload += MakeJWTAssertion(assertion_header, assertion_payload,
109  svc_acct_private_key_pem);
110  payload_ = std::move(payload);
111 
112  request_builder.AddHeader(
113  "Content-Type: application/x-www-form-urlencoded");
114  request_ = request_builder.BuildRequest();
115  }

117  std::string AuthorizationHeader() override {
118  std::unique_lock<std::mutex> lk(mu_);
119  cv_.wait(lk, [this]() { return Refresh(); });
120  return authorization_header_;
121  }
122 
123  private:
124  std::string MakeJWTAssertion(storage::internal::nl::json const& header,
125  storage::internal::nl::json const& payload,
126  std::string const& pem_contents) {
127  using storage::internal::OpenSslUtils;
128  std::string encoded_header =
129  OpenSslUtils::UrlsafeBase64Encode(header.dump());
130  std::string encoded_payload =
131  OpenSslUtils::UrlsafeBase64Encode(payload.dump());
132  std::string encoded_signature =
133  OpenSslUtils::UrlsafeBase64Encode(OpenSslUtils::SignStringWithPem(
134  encoded_header + '.' + encoded_payload, pem_contents,
135  JwtSigningAlgorithms::RS256));
136  return encoded_header + '.' + encoded_payload + '.' + encoded_signature;
137  }
138 
139  bool Refresh() {
140  namespace nl = storage::internal::nl;
141  if (std::chrono::system_clock::now() < expiration_time_) {
142  return true;
143  }
144 
145  // TODO(#516) - use retry policies to refresh the credentials.
146  auto response = request_.MakeRequest(payload_);
147  if (200 != response.status_code) {
148  return false;
149  }
150 
151  nl::json access_token = nl::json::parse(response.payload);
152  // Response should have the attributes "access_token", "expires_in", and
153  // "token_type".
154  std::string header =
155  "Authorization: " +
156  access_token["token_type"].get_ref<std::string const&>() + " " +
157  access_token["access_token"].get_ref<std::string const&>();
158  auto expires_in = std::chrono::seconds(access_token["expires_in"]);
159  auto new_expiration = std::chrono::system_clock::now() + expires_in -
160  GoogleOAuthTokenExpirationSlack();
161  // Do not update any state until all potential exceptions are raised.
162  authorization_header_ = std::move(header);
163  expiration_time_ = new_expiration;
164  return true;
165  }

170  std::condition_variable cv_;
171  std::string authorization_header_;
172  std::chrono::system_clock::time_point expiration_time_;
173  ClockType clock_;
174 };
