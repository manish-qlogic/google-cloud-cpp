69 template <typename HttpRequestBuilderType =
70  storage::internal::CurlRequestBuilder,
71  typename ClockType = std::chrono::system_clock>
72 class ServiceAccountCredentials : public Credentials {
73  public:
74  explicit ServiceAccountCredentials(std::string const& content,
75  std::string const& source)
76  : ServiceAccountCredentials(content, source,
77  GoogleOAuthRefreshEndpoint()) {}
78 
79  explicit ServiceAccountCredentials(std::string const& content,
80  std::string const& source,
81  std::string const& default_token_uri)
82  : expiration_time_(), clock_() {
83  namespace nl = storage::internal::nl;
84  auto info =
85  ParseServiceAccountCredentials(content, source, default_token_uri);
86  // Below, we construct a JWT refresh request used to obtain an access token.
87  // The structure of a JWT is defined in RFC 7519 (see
88  // https://tools.ietf.org/html/rfc7519), and Google-specific JWT validation
89  // logic is further described at:
90  // https://cloud.google.com/endpoints/docs/frameworks/java/troubleshoot-jwt
91  nl::json assertion_header = {
92  {"alg", "RS256"}, {"kid", info.private_key_id}, {"typ", "JWT"}};
93 
94  std::string scope = GoogleOAuthScopeCloudPlatform();
95 
96  // As much as possible do the time arithmetic using the std::chrono types,
97  // convert to longs only when we are dealing with timestamps since the
98  // epoch.
99  auto now = clock_.now();
100  auto expiration = now + GoogleOAuthAccessTokenLifetime();
101  auto now_from_epoch =
102  static_cast<long>(std::chrono::system_clock::to_time_t(now));
103  auto expiration_from_epoch =
104  static_cast<long>(std::chrono::system_clock::to_time_t(expiration));
105  nl::json assertion_payload = {
106  {"iss", info.client_email},
107  {"scope", scope},
108  {"aud", info.token_uri},
109  {"iat", now_from_epoch},
110  // Resulting access token should be expire after one hour.
111  {"exp", expiration_from_epoch}};
112 
113  HttpRequestBuilderType request_builder(
114  std::move(info.token_uri),
115  storage::internal::GetDefaultCurlHandleFactory());
116  // This is the value of grant_type for JSON-formatted service account
117  // keyfiles downloaded from Cloud Console.
118  std::string payload("grant_type=");
119  payload +=
120  request_builder
121  .MakeEscapedString("urn:ietf:params:oauth:grant-type:jwt-bearer")
122  .get();
123  payload += "&assertion=";
124  payload +=
125  MakeJWTAssertion(assertion_header, assertion_payload, info.private_key);
126  payload_ = std::move(payload);
127 
128  request_builder.AddHeader(
129  "Content-Type: application/x-www-form-urlencoded");
130  request_ = request_builder.BuildRequest();
131  }

133  std::string AuthorizationHeader() override {
134  std::unique_lock<std::mutex> lk(mu_);
135  cv_.wait(lk, [this]() { return Refresh().ok(); });
136  return authorization_header_;
137  }
138 
139  private:
140  std::string MakeJWTAssertion(storage::internal::nl::json const& header,
141  storage::internal::nl::json const& payload,
142  std::string const& pem_contents) {
143  using storage::internal::OpenSslUtils;
144  std::string encoded_header =
145  OpenSslUtils::UrlsafeBase64Encode(header.dump());
146  std::string encoded_payload =
147  OpenSslUtils::UrlsafeBase64Encode(payload.dump());
148  std::string encoded_signature =
149  OpenSslUtils::UrlsafeBase64Encode(OpenSslUtils::SignStringWithPem(
150  encoded_header + '.' + encoded_payload, pem_contents,
151  JwtSigningAlgorithms::RS256));
152  return encoded_header + '.' + encoded_payload + '.' + encoded_signature;
153  }
154 
155  storage::Status Refresh() {
156  namespace nl = storage::internal::nl;
157  if (std::chrono::system_clock::now() < expiration_time_) {
158  return storage::Status();
159  }
160 
161  // TODO(#516) - use retry policies to refresh the credentials.
162  auto response = request_.MakeRequest(payload_);
163  if (response.status_code >= 300) {
164  return storage::Status(response.status_code, std::move(response.payload));
165  }
166 
167  nl::json access_token = nl::json::parse(response.payload, nullptr, false);
168  if (access_token.is_discarded() or
169  access_token.count("access_token") == 0U or
170  access_token.count("expires_in") == 0U or
171  access_token.count("token_type") == 0U) {
172  return storage::Status(
173  response.status_code, std::move(response.payload),
174  "Could not find all required fields in response (access_token,"
175  " expires_in, token_type).");
176  }
177  // Response should have the attributes "access_token", "expires_in", and
178  // "token_type".
179  std::string header =
180  "Authorization: " + access_token.value("token_type", "") + " " +
181  access_token.value("access_token", "");
182  auto expires_in =
183  std::chrono::seconds(access_token.value("expires_in", int(0)));
184  auto new_expiration = std::chrono::system_clock::now() + expires_in -
185  GoogleOAuthAccessTokenExpirationSlack();
186  // Do not update any state until all potential exceptions are raised.
187  authorization_header_ = std::move(header);
188  expiration_time_ = new_expiration;
189  return storage::Status();
190  }

198  ClockType clock_;
