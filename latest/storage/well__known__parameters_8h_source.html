34 template <typename P, typename T>
35 class WellKnownParameter {
36  public:
37  WellKnownParameter() : value_{} {}
38  explicit WellKnownParameter(T&& value) : value_(std::forward<T>(value)) {}
39  explicit WellKnownParameter(T const& value) : value_(value) {}
40 
41  char const* parameter_name() const { return P::well_known_parameter_name(); }
42  bool has_value() const { return value_.has_value(); }
43  T const& value() const { return value_.value(); }
44 
45  private:
46  google::cloud::optional<T> value_;
47 };
48 
49 template <typename P, typename T>
50 std::ostream& operator<<(std::ostream& os,
51  WellKnownParameter<P, T> const& rhs) {
52  if (rhs.has_value()) {
53  return os << rhs.parameter_name() << "=" << rhs.value();
54  }
55  return os << rhs.parameter_name() << "=<not set>";
56 }
