240 template <typename Generator>
241 static EncryptionKeyData CreateKeyFromGenerator(Generator& gen) {
242  static_assert(
243  std::numeric_limits<unsigned char>::digits == 8,
244  "The Google Cloud Storage C++ library is only supported on platforms\n"
245  "with 8-bit chars. Please file a bug on\n"
246  " https://github.com/GoogleCloudPlatform/google-cloud-cpp/issues\n"
247  "describing your platform details to request support for it.");
248  constexpr int kKeySize = 256 / std::numeric_limits<unsigned char>::digits;
249 
250  constexpr auto minchar = (std::numeric_limits<char>::min)();
251  constexpr auto maxchar = (std::numeric_limits<char>::max)();
252  std::uniform_int_distribution<int> uni(minchar, maxchar);
253  std::string key(static_cast<std::size_t>(kKeySize), ' ');
254  std::generate_n(key.begin(), key.size(),
255  [&uni, &gen] { return static_cast<char>(uni(gen)); });
256  return EncryptionDataFromBinaryKey(key);
257 }
