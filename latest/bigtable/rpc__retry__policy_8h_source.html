30 struct SafeGrpcRetry {
31  // Sometimes we need to use google::protobuf::rpc::Status, in which case this
32  // is a easier function
33  static inline bool IsTransientFailure(grpc::StatusCode code) {
34  return code == grpc::StatusCode::ABORTED or
35  code == grpc::StatusCode::UNAVAILABLE or
36  code == grpc::StatusCode::DEADLINE_EXCEEDED;
37  }
38 
39  static inline bool IsOk(grpc::Status const& status) { return status.ok(); }
40  static inline bool IsTransientFailure(grpc::Status const& status) {
41  return IsTransientFailure(status.error_code());
42  }
43  static inline bool IsPermanentFailure(grpc::Status const& status) {
44  return not IsOk(status) and not IsTransientFailure(status);
45  }
46 };
