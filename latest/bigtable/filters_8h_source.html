518  template <typename... FilterTypes>
519  static Filter Chain(FilterTypes&&... stages) {
520  // This ugly thing provides a better compile-time error message than
521  // just letting the compiler figure things out 3 levels deep
522  // as it recurses on append_types().
523  static_assert(
524  internal::conjunction<
525  std::is_convertible<FilterTypes, Filter>...>::value,
526  "The arguments passed to Chain(...) must be convertible to Filter");
527  Filter tmp;
528  auto& chain = *tmp.filter_.mutable_chain();
529  std::initializer_list<Filter> list{std::forward<FilterTypes>(stages)...};
530  for (Filter const& f : list) {
531  *chain.add_filters() = f.as_proto();
532  }
533  return tmp;
534  }
